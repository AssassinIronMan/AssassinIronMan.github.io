canvasCtx.fillRect(0, 0, width, height); canvasCtx.lineWidth = 2; canvasCtx.strokeStyle = '#007bff'; canvasCtx.beginPath(); const sliceWidth = width / analyser.fftSize; let x = 0; for (let i = 0; i < analyser.fftSize; i++) { const v = dataArray[i] / 128.0; // Normalize const y = v * height / 2; if (i === 0) { canvasCtx.moveTo(x, y); } else { canvasCtx.lineTo(x, y); } x += sliceWidth; } canvasCtx.lineTo(canvas.width, canvas.height / 2); canvasCtx.stroke(); } drawWaveform(); } function detectPops() { analyser.getByteTimeDomainData(dataArray); // Convert raw amplitude to decibels let rms = 0; for (let i = 0; i < dataArray.length; i++) { rms += Math.pow(dataArray[i] - 128, 2); } rms = Math.sqrt(rms / dataArray.length); const decibels = 20 * Math.log10(rms / 128); // Check if the decibel level crosses the threshold if (decibels > volumeThresholdDB) { popCount++; silenceStartTime = audioContext.currentTime; // Reset silence timer showPopup('Pops detected: ' + popCount); if (customPopSound.src) { customPopSound.play(); } } else if (audioContext.currentTime - silenceStartTime >= silenceDuration) { // If silence has lasted for the specified duration showPopup('Popcorn DONE! Total Pops: ' + popCount); stopDetection(); // Stop detection after showing the message } requestAnimationFrame(detectPops); } function stopDetection() { if (audioContext) { audioContext.close(); document.getElementById('status').textContent = 'Detection stopped.'; } } function updateSettings() { volumeThresholdDB = document.getElementById('volumeThreshold').value; popInterval = document.getElementById('popInterval').value; fftSize = document.getElementById('sensitivity').value; // Prevent simulation if it's within the interval } isSimulating = true; document.getElementById('status').textContent = "Simulating Pop..."; // Simulate a noise spike const simulatedPopDuration = 0.1; showPopup('Popcorn DONE!'); if (customPopSound.src) { customPopSound.play(); } isSimulating = false; lastSimulatedPopTime = currentTime; // Update last simulated pop time }, simulatedPopDuration * 1000); // Convert seconds to milliseconds } function handleFileSelect(event) { const file = event.target.files[0]; if (file) { customPopSound.src = URL.createObjectURL(file); } }
